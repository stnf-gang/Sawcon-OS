misc_functions.o is a 16 bit library

functions:

void putc(char __c)
    info:
        - prints a character using BIOS interrupts
        - expects the screen to be in text mode

void puts(char *__s)
    info:
        - prints a NULL terminated ASCII string using putc

void __cdecl puts_len(char *__s, int _length)
    info:
        - printf a NULL terminated ASCII string up to a certain number of characters
